\frame
{
    \frametitle{Bonus: My productivity tips - cont'd}
    \begin{itemize}
	\item Project folder structure:
	\begin{itemize}	
	\item \textcolor{darkblue}{src/} Source code, must have: README.md | .gitignore | config.ini
	\item \textcolor{darkblue}{doc/} Manuscripts, presentations, related articles
	\item \textcolor{darkblue}{data/} Data used in the project
	\end{itemize}
	\bigskip
	\item Backing up \textit{essentials}:
	\begin{itemize}	
	\item Make \textcolor{darkblue}{src} as an upstream git repository residing @ \textcolor{darkblue}{Dropbox}
	\item Make \textcolor{darkblue}{src} publicly available @ \textcolor{darkblue}{Github} (unless needs to be confidential)
	\item Keep \textcolor{darkblue}{doc} in \textcolor{darkblue}{Dropbox} and make it symbolic link in the project
	\end{itemize}
    \end{itemize}
}
